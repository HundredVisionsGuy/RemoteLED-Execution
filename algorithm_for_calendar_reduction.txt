Algorithm for checking for A or B day.
GET datetime for today
GET current_year from datetime
TRY
    READ calendar
EXCEPT File not found
    CALL get_calendar() with current_year


DEFINE get_calendar()
    GET calendar
    SET same_year to False
    SET calendar_list to empty list
    FOR EACH line in calendar
        IF same_year
            APPEND line to calendar_list
        ELSE IF "DTSTART" in line
            GET datetime
            IF datetime.year < current_year
                CONTINUE
            ELSE:
                SET same_year to True
            END IF
        END IF
    END FOR
    WRITE calendar_list to calendar
END DEFINITION

DEFINE get_day(calendar)
    GET today
    GET current_day
    IF current_day == Saturday or Sunday
        RETURN "weekend"
    FOR EACH event in calendar
        GET summary
        IF "A day" in summary or "B day" in summary
            GET start_date
            IF start_date == today
                RETURN summary
            ELSE IF "FREQ=DAILY" in RRULE
                GET until_date
                GET interval (should be 2)
                IF today > start_date and today <= until_date
                    SET days_since to today - start_date
                    IF days_since MOD interval == 0
                        RETURN summary
                    ENDIF
                ENDIF
            ENDIF
        ENDIF
    ENDFOR
